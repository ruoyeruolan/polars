29impl Buffer<'_> {
30    pub fn into_series(self) -> Series {
31        let mut s = self.buf.into_series();
32        s.rename(PlSmallStr::from_str(self.name));
33        s
34    }

36    #[inline]
37    pub(crate) fn add(&mut self, value: &Value) -> PolarsResult<()> {
38        use AnyValueBuffer::*;
39        match &mut self.buf {
40            Boolean(buf) => {
41                match value {
42                    Value::Static(StaticNode::Bool(b)) => buf.append_value(*b),
43                    _ => buf.append_null(),
44                }
45                Ok(())
46            },
47            Int32(buf) => {
48                let n = deserialize_number::<i32>(value);
49                match n {
50                    Some(v) => buf.append_value(v),
51                    None => buf.append_null(),
52                }
53                Ok(())
54            },
55            Int64(buf) => {
56                let n = deserialize_number::<i64>(value);
57                match n {
58                    Some(v) => buf.append_value(v),
59                    None => buf.append_null(),
60                }
61                Ok(())
62            },
63            UInt64(buf) => {
64                let n = deserialize_number::<u64>(value);
65                match n {
66                    Some(v) => buf.append_value(v),
67                    None => buf.append_null(),
68                }
69                Ok(())
70            },
71            UInt32(buf) => {
72                let n = deserialize_number::<u32>(value);
73                match n {
74                    Some(v) => buf.append_value(v),
75                    None => buf.append_null(),
76                }
77                Ok(())
78            },
79            Float32(buf) => {
80                let n = deserialize_number::<f32>(value);
81                match n {
82                    Some(v) => buf.append_value(v),
83                    None => buf.append_null(),
84                }
85                Ok(())
86            },
87            Float64(buf) => {
88                let n = deserialize_number::<f64>(value);
89                match n {
90                    Some(v) => buf.append_value(v),
91                    None => buf.append_null(),
92                }
93                Ok(())
94            },
95
96            String(buf) => {
97                match value {
98                    Value::String(v) => buf.append_value(v),
99                    _ => buf.append_null(),
100                }
101                Ok(())
102            },
103            #[cfg(feature = "dtype-datetime")]
104            Datetime(buf, _, _) => {
105                let v = deserialize_datetime::<Int64Type>(value);
106                buf.append_option(v);
107                Ok(())
108            },
109            #[cfg(feature = "dtype-date")]
110            Date(buf) => {
111                let v = deserialize_datetime::<Int32Type>(value);
112                buf.append_option(v);
113                Ok(())
114            },
115            All(dtype, buf) => {
116                let av = deserialize_all(value, dtype, self.ignore_errors)?;
117                buf.push(av);
118                Ok(())
119            },
120            Null(builder) => {
121                builder.append_null();
122                Ok(())
123            },
124            _ => panic!("unexpected dtype when deserializing ndjson"),
125        }
126    }
127    pub fn add_null(&mut self) {
128        self.buf.add(AnyValue::Null).expect("should not fail");
129    }

131pub(crate) fn init_buffers(
132    schema: &Schema,
133    capacity: usize,
134    ignore_errors: bool,
135) -> PolarsResult<PlIndexMap<BufferKey, Buffer>> {
136    schema
137        .iter()
138        .map(|(name, dtype)| {
139            let av_buf = (dtype, capacity).into();
140            let key = KnownKey::from(name.as_str());
141            Ok((
142                BufferKey(key),
143                Buffer {
144                    name,
145                    buf: av_buf,
146                    ignore_errors,
147                },
148            ))
149        })
150        .collect()
151}

153fn deserialize_number<T: NativeType + NumCast>(value: &Value) -> Option<T> {
154    match value {
155        Value::Static(StaticNode::F64(f)) => num_traits::cast(*f),
156        Value::Static(StaticNode::I64(i)) => num_traits::cast(*i),
157        Value::Static(StaticNode::U64(u)) => num_traits::cast(*u),
158        Value::Static(StaticNode::Bool(b)) => num_traits::cast(*b as i32),
159        _ => None,
160    }
161}

163#[cfg(feature = "dtype-datetime")]
164fn deserialize_datetime<T>(value: &Value) -> Option<T::Native>
165where
166    T: PolarsNumericType,
167    DatetimeInfer<T>: TryFromWithUnit<Pattern>,
168{
169    let val = match value {
170        Value::String(s) => s,
171        _ => return None,
172    };
173    infer_pattern_single(val).and_then(|pattern| {
174        match DatetimeInfer::try_from_with_unit(pattern, Some(TimeUnit::Microseconds)) {
175            Ok(mut infer) => infer.parse(val),
176            Err(_) => None,
177        }
178    })
179}

181fn deserialize_all<'a>(
182    json: &Value,
183    dtype: &DataType,
184    ignore_errors: bool,
185) -> PolarsResult<AnyValue<'a>> {
186    let out = match json {
187        Value::Static(StaticNode::Bool(b)) => AnyValue::Boolean(*b),
188        Value::Static(StaticNode::I64(i)) => AnyValue::Int64(*i),
189        Value::Static(StaticNode::U64(u)) => AnyValue::UInt64(*u),
190        Value::Static(StaticNode::F64(f)) => AnyValue::Float64(*f),
191        Value::Static(StaticNode::Null) => AnyValue::Null,
192        Value::String(s) => AnyValue::StringOwned(s.as_ref().into()),
193        Value::Array(arr) => {
194            let Some(inner_dtype) = dtype.inner_dtype() else {
195                if ignore_errors {
196                    return Ok(AnyValue::Null);
197                }
198                polars_bail!(ComputeError: "expected dtype '{}' in JSON value, got dtype: Array\n\nEncountered value: {}", dtype, json);
199            };
200            let vals: Vec<AnyValue> = arr
201                .iter()
202                .map(|val| deserialize_all(val, inner_dtype, ignore_errors))
203                .collect::<PolarsResult<_>>()?;
204            let s =
205                Series::from_any_values_and_dtype(PlSmallStr::EMPTY, &vals, inner_dtype, false)?;
206            AnyValue::List(s)
207        },
208        #[cfg(feature = "dtype-struct")]
209        Value::Object(doc) => {
210            if let DataType::Struct(fields) = dtype {
211                let document = &**doc;
212
213                let vals = fields
214                    .iter()
215                    .map(|field| {
216                        if let Some(value) = document.get(field.name.as_str()) {
217                            deserialize_all(value, &field.dtype, ignore_errors)
218                        } else {
219                            Ok(AnyValue::Null)
220                        }
221                    })
222                    .collect::<PolarsResult<Vec<_>>>()?;
223                AnyValue::StructOwned(Box::new((vals, fields.clone())))
224            } else {
225                if ignore_errors {
226                    return Ok(AnyValue::Null);
227                }
228                polars_bail!(
229                    ComputeError: "expected {} in json value, got object", dtype,
230                );
231            }
232        },
233        #[cfg(not(feature = "dtype-struct"))]
234        val => AnyValue::StringOwned(format!("{:#?}", val).into()),
235    };
236    Ok(out)
237}
